cmake_minimum_required(VERSION 3.20)
project(BlackHoleRayTracer)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Tell CMake where your headers live
include_directories(${PROJECT_SOURCE_DIR}/Header)

# Source files
file(GLOB SRC_FILES src/*.cpp)   # All cpp files in src/

# Create executable
add_executable(BlackHoleRayTracer ${SRC_FILES} "src/Graphics.cpp")

# Find packages via vcpkg
find_package(glfw3 REQUIRED)
find_package(glm REQUIRED)
find_package(glad REQUIRED)
find_package(OpenGL REQUIRED)

# Link libraries
target_link_libraries(BlackHoleRayTracer
    glfw
    glad::glad
    glm::glm
    OpenGL::GL
)
